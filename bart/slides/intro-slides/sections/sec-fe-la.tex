\begin{frame}{Linear algebra in \ttnc{BART}}
	\begin{block}{Sparse matrix-vector product based computations}
		\begin{itemize}
			\item Current implementation of \ttt{BART} assembles global matrices and utilize sparse matrix-vector product in linear algebraic solvers.
			\begin{itemize}
				\item Easy implementation.
				\item High computational efficiency with (bi/tri-) linear elements.
			\end{itemize}
		\end{itemize}
	\end{block}
	\begin{block}{\ttnc{BART} is interfaced with \ttnc{PETSc}}
		\begin{itemize}
			\item Most \ttt{PETSc} solvers/preconditioners wrapped in \ttt{deal.II} are used in \ttt{PreconditionerSolver} class of \ttt{BART}
			\begin{itemize}
				\item Direct solver: parallel direct solver \ttt{MUMPS}
				\item Iterative solvers and preconditioners
			\end{itemize}
			\item Performance remedy: as solving will happen multiple times due to source/power iterations, we initialize the preconditioning/factorization only once then preconditioning/factorization matrices will be stored for reuse.
		\end{itemize}
	\end{block}
\end{frame}
